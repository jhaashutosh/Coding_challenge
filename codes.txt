#include <iostream>
using namespace std;
void reverse(int a[],int start,int end)
{
    int x=((end-start)/2);
    if(start==0)
    {
        for(int i=start;i<=x;i++)
        {
            int temp=a[i];
            a[i]=a[end-i];
            a[end-i]=temp;
        }
        
    }
    else{
        for(int i=0;i<=x;i++)
        {
            int temp=a[start];
            a[start]=a[end];
            a[end]=temp;
            start++;
            end--;
        }
    }
    
}
int main() {
	int n;
	cin>>n;
	int a[n];
	for(int i=0;i<n;i++)
	{
	    cin>>a[i];
	}
	int d;
	cin>>d;
	reverse(a,0,d-1);
	for(int i=0;i<n;i++)
	{
	    cout<<a[i]<<" ";
	}
	cout<<endl;
	reverse(a,d,n-1);
	for(int i=0;i<n;i++)
	{
	    cout<<a[i]<<" ";
	}
	cout<<endl;
	reverse(a,0,n-1);
	for(int i=0;i<n;i++)
	{
	    cout<<a[i]<<" ";
	}
	cout<<endl;
	
}
