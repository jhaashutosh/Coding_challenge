//Find the Rotation Count in Rotated Sorted array

class Solution{
public:	
	int findKRotation(int arr[], int n) {
	   
	    int count=1;
	    
	   for(int i=0;i<n-1;i++)
	   {
	       if(arr[i]>arr[i+1])
	       {
	           return count;
	       }
	       
	      else
	       count++;
	   }
	   return 0;
	}

};
