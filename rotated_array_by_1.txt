clockwise rotate an array by 1.
#include <iostream>

using namespace std;

int main()
{
    int arr[]={1,2,3,4,5,6};
    int n=6;
    int x=arr[n-1];
    // cout<<x<<endl;
    for(int i=n-1;i>0;i--)
    {
        arr[i]=arr[i-1];
    }
    arr[0]=x;
    for(int i=0;i<n;i++)
    {
        cout<<arr[i]<<" ";
    }

    return 0;
}
