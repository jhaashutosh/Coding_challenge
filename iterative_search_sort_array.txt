//iterative approach of seaching an element in a sorted + rotated array.....

#include <iostream>

using namespace std;


int main()
{
    int a[]={3,4,5,6,7,1,2};//2 anti-clock rotations required....//3 clock wise
    
  int l=0,h=6;
  int key;
  cin>>key;
  
  while(l<=h)
  {
      int mid=(h+l)/2;
   
      
      if(a[mid]==key)
     {
         cout<<mid;
         return mid;
     }
      else if(a[mid]>a[l])
      {
          if(key>=a[l]&&key<a[mid])
          h=mid-1;
          else 
          l=mid+1;
      }
      else if(a[mid]<a[h])
      {
          if(a[mid]<key&&key<=a[h])
          l=mid+1;
          else
          h=mid-1;
      }
      else 
      {
         return -1;
      }
  }
    

    return 0;
}
